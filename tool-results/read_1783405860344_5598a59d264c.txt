   532→// Strip ```markdown ... ``` code fences that the LLM may wrap around the output
   533→function stripCodeFences(content: string): string {
   534→  let cleaned = content.trim();
   535→  // Remove leading ```markdown or ```md or ```
   536→  cleaned = cleaned.replace(/^```(?:markdown|md)?\s*\n/i, "");
   537→  // Remove trailing ```
   538→  cleaned = cleaned.replace(/\n```\s*$/i, "");
   539→  return cleaned.trim();
   540→}
   541→
   542→function renderMarkdownToHtml(content: string): string {
   543→  // Simple markdown-to-HTML for the print window (no react-markdown available there)
   544→  const cleaned = stripCodeFences(content);
   545→  let html = cleaned
   546→    .replace(/^#### (.+)/gm, "<h4>$1</h4>")
   547→    .replace(/^### (.+)/gm, "<h3>$1</h3>")
   548→    .replace(/^## (.+)/gm, "<h2>$1</h2>")
   549→    .replace(/^# (.+)/gm, "<h1>$1</h1>")
   550→    .replace(/\*\*(.+?)\*\*/g, "<strong>$1</strong>")
   551→    .replace(/\*(.+?)\*/g, "<em>$1</em>")
   552→    .replace(/\[([^\]]+)\]\(([^)]+)\)/g, '<a href="$2">$1</a>')
   553→    .replace(/^---$/gm, "<hr>")
   554→    .replace(/^\s*[-*]\s(.+)/gm, "<li>$1</li>")
   555→    .replace(/(<li>[\s\S]*?<\/li>)(?=\n\n|\n##|\n#|\Z)/g, "<ul>$1</ul>");
   556→
   557→  // Convert tables
   558→  const tableLines = cleaned.split("\n").filter((l) => l.startsWith("| "));
   559→  if (tableLines.length > 0) {
   560→    html = html.replace(/^\|(.+)\|$/gm, (match) => {
   561→      const cells = match.split("|").filter((c) => c.trim());
   562→      const tds = cells.map((c) => `<td>${c.trim()}</td>`).join("");
   563→      return `<tr>${tds}</tr>`;
   564→    });
   565→    html = html.replace(/(<tr>[\s\S]*?<\/tr>)(?=\n\n|\n##|\n#|\Z)/g, "<table>$1</table>");
   566→  }
   567→
   568→  return html;
   569→}
